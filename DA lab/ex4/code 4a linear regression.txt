import pandas as pd
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score

# Load dataset
d = pd.read_csv('exam4.csv')

# Define features and target
X = d[['Age']]        # independent variable
y = d['Salary']      # dependent variable (1D)

# Split data
xtr, xte, ytr, yte = train_test_split(X, y, test_size=0.3, random_state=42)

# Train model
m = LinearRegression()
m.fit(xtr, ytr)

# Predictions
yp = m.predict(xte)

# Results
print("Slope:", m.coef_[0])
print("Intercept:", m.intercept_)
print("MSE:", mean_squared_error(yte, yp))
print("R2 Score:", r2_score(yte, yp))

# --- TRAINING GRAPH ---
plt.scatter(xtr, ytr, color='blue', label='Training Data')
plt.plot(xtr, m.predict(xtr), color='red', label='Regression Line')
plt.xlabel("Age")
plt.ylabel("Salary")
plt.title("Training Data Regression")
plt.legend()
plt.show()

# --- TEST GRAPH ---
plt.scatter(xte, yte, color='green', label='Test Data')
plt.plot(xte, yp, color='red', label='Regression Line')
plt.xlabel("Age")
plt.ylabel("Salary")
plt.title("Test Data Regression")
plt.legend()
plt.show()
