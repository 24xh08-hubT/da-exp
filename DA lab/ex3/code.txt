from efficient_apriori import apriori
import pandas as pd
d=pd.read_csv("ex3dataset.csv",names=["product"],header=None);print(d,"\n")
i,r=apriori([x.split(",") for x in d["product"]],0.3,0.6)
print("Itemset\t\tSupport")
for a in i:
 for b in i[a]:print(b,"\t",i[a][b])
print("\nAntecedent\tConsequent\tSupport\tConfidence\tLift")
for x in r:print(x.lhs,"==>",x.rhs,"\t",x.support,"\t",x.confidence,"\t",x.lift)
